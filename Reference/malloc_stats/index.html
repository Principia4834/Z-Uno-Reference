malloc_stats()

Отображает информацию об используемой памяти malloc() Отображает с помощью Serial, предварительно нужно иницилизировать с помощью begin() malloc_stats(fd)
 fd: указывает какой Serial использовать от 0 до 2. nothing 
		
void setup() {
	Serial0.begin(115200);//Иницилизируем Serial0 для вывода информации
}

void loop() {
	malloc_stats(0);//Указываем что в Serial0 нужно выводить и выводим
	Serial0.println();
	delay(1000);
}
