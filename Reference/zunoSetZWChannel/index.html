zunoSetZWChannel()

Присваивает каналу zwave канал. Сначала добавте канал с помощью zunoAddChannel, а потом установите zwave канал с помощью zunoSetZWChannel. 
zunoSetZWChannel(ch, zw_channel)
  • ch канал от 0x0 до 31 после первого использовании zunoAddChannel у вас есть канал 0x0, после второго у вас уже 0x0 и 0x1...
  • zw_channel - zwave канал может формироваться двумя способами: кластерно или не кластерно.В случае если не используеться кластаризация то каждый zwave канал уникальный, так же к каждому первому каналу одного типа добавляеться еще и ZWAVE_CHANNEL_MAPPED_BIT - zunoSetZWChannel(0x0, 0x01 | ZWAVE_CHANNEL_MAPPED_BIT). В случае использования кластаризации все тоже самое, но можно каждому каналу тип которого есть в одном только количстве то присваиваеться zw_channel нуллю - zunoSetZWChannel(0x0, 0x0).
 В Примере №4 zunoStartDeviceConfiguration используеться zunoSetZWChannel без класторизации, переделаем уже с кластеризацией: 
// Указываем что будем использовать. В нашем случае это два команд класса, если их не указать то код для этих команд классов не собереться.
ZUNO_ENABLE(WITH_CC_SWITCH_MULTILEVEL WITH_CC_BASIC)

byte dimmerValue = 99;

void setup() {
  if(zunoStartDeviceConfiguration()) {//Провераем включенно устройство в сеть или нет, если не включенно то настраиваем каналлы.
    zunoAddChannel(ZUNO_SWITCH_MULTILEVEL_CHANNEL_NUMBER, 0x0, 0x0);//Добавляем в канал тип SWITCH_MULTILEVEL, без каких доп параметров
    zunoSetZWChannel(0x0, 0x1 | ZWAVE_CHANNEL_MAPPED_BIT);//Указываем zwave канал
    zunoCommitCfg();//Применяем настройки - настройки сохраняються во флеш поэтому уже включенное устройство не нужно повторно настраивать.
  }
  zunoAppendChannelHandler( 0,  1, CHANNEL_HANDLER_SINGLE_VALUEMAPPER, (void*)&dimmerValue);//Указываем адресс для переменной - что бы код мог ее использовать. Это нужно устанавливать каждый раз.

  pinMode(LED_BUILTIN, OUTPUT);
}

void loop() {
  digitalWrite(LED_BUILTIN, HIGH);
  delay(dimmerValue*10);
  digitalWrite(LED_BUILTIN, LOW);
  delay(dimmerValue*10);
}

  if(zunoStartDeviceConfiguration()) {//Провераем включенно устройство в сеть или нет, если не включенно то настраиваем каналлы.
    zunoAddChannel(ZUNO_SWITCH_MULTILEVEL_CHANNEL_NUMBER, 0x0, 0x0);//Добавляем в канал тип SWITCH_MULTILEVEL, без каких доп параметров

    zunoAddChannel(ZUNO_SWITCH_MULTILEVEL_CHANNEL_NUMBER, 0x0, 0x0);//Добавляем в канал тип SWITCH_MULTILEVEL, без каких доп параметров
    zunoSetZWChannel(0x1, 0x2);//Указываем zwave канал

  }
