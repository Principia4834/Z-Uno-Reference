zunoAssociationGroupName()

Позволит для групп установить свое имя иначе будет использованно имя группы типа "User group 01", "User group 02"... zunoAssociationGroupName(groupIndex)
 groupIndex: индекс группы асоциации, начинаеться он с CTRL_GROUP_1... указатель на Си строку распаложенную в постоянной памяти и длинной не более 42 байта 
		

ZUNO_SETUP_ASSOCIATIONS(ZUNO_ASSOCIATION_GROUP_SET_VALUE, ZUNO_ASSOCIATION_GROUP_SET_VALUE_AND_DIM)

void setup() {
}

const char *zunoAssociationGroupName(uint8_t groupIndex) {
	const char				*tmp;

	switch (groupIndex) {
		case CTRL_GROUP_1:
			tmp = "ZUNO_ASSOCIATION_GROUP_SET_VALUE";
			break;
		case CTRL_GROUP_2:
			tmp = "ZUNO_ASSOCIATION_GROUP_SET_VALUE_AND_DIM";
			break;
		default:
			tmp = NULL;
			break;
	}
	return (tmp);
}
