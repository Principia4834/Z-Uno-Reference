zunoPTIConfigUART()

This function allows you to configuration Packet Trace Interface. 
zunoPTIConfigUART(tx_pin, baud)
 tx_pin: Any Z-Uno pin. baud: Output baudrate; minimal baud is 230400; maximum baud is 1.6MBit bool: true - if success. 
		
#define MY_SERIAL				Serial0

#define PTI_PIN					TX2// TX2 is default USB CDC pin
#define PTI_BAUDRATE			230400

void setup() {
	MY_SERIAL.begin(115200);
	delay(1000);
	MY_SERIAL.print("\n\nEnable Packet Trace Interface on pin#");
	MY_SERIAL.print(PTI_PIN);
	MY_SERIAL.print(" BAUD:");
	MY_SERIAL.print(PTI_BAUDRATE);
	MY_SERIAL.print(" ");
	if(zunoPTIConfigUART(PTI_PIN, PTI_BAUDRATE) == true)
		Serial0.println("[Done]");
	else
		Serial0.println("[Failed]");
}
