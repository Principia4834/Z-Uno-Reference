Type ZUNO_HANDLER_SYSTIMER

Данный тип хендлера позволяет привязать свою функцию к системному тайммеру. Вызов будет происходить каждыйе 10мс. В данном событие `sub_type` устанавливаеться в 0 обычно, но можно использовать любое число. handlerSysTimer(ms)
 ms: number of milliseconds since the Z-Uno board began running the current program. nothing 
		
// С помощью события "системный таймер" - делаем мигание диодом

byte state = HIGH;
byte dimmerValue=100;
size_t glob;

// Функция обработчика
void handlerSysTimer(size_t ms) {
	ms = millis();
	if (ms >= glob)//Провераем нужно ли управлять пином
	{
		glob = ms + dimmerValue*10;
		digitalWrite(LED_BUILTIN, state);
		state = state == HIGH ? LOW : HIGH;
	}
	
}

void setup() {
	pinMode(LED_BUILTIN, OUTPUT);
	zunoAttachSysHandler(ZUNO_HANDLER_SYSTIMER, 0x0, (void *)&handlerSysTimer);//Устанавливаем в событие свой обработчик который будет вызываться каждые 10мс
	glob = millis() + dimmerValue*10;
}

void loop() {

}
